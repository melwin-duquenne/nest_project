Étape 2 — Dockerfile multi-stage

Créer Dockerfile à la racine.

Un Dockerfile multi-stage a deux étapes :

  1. Builder — image complète avec devDependencies, compile TypeScript (npm run build)
  2. Runner — image légère avec prod dependencies uniquement, copie le dist/ depuis builder
# ───── Étape 1 : Builder ─────
FROM node:20-alpine AS builder
WORKDIR /app

COPY package*.json ./
RUN npm ci

COPY . .
RUN npm run build

# ───── Étape 2 : Runner ─────
FROM node:20-alpine AS runner
WORKDIR /app

COPY package*.json ./
RUN npm ci --only=production && npm cache clean --force

COPY --from=builder /app/dist ./dist

# Sécurité : utilisateur non-root
RUN addgroup -S appgroup && adduser -S appuser -G appgroup
USER appuser

ENV NODE_ENV=production
EXPOSE 3000

HEALTHCHECK --interval=30s --timeout=5s --start-period=10s \
  CMD wget -qO- http://localhost:3000/api/health || exit 1

CMD ["node", "dist/main.js"]

Tester le build :

docker build -t taskflow-api:local .
docker images taskflow-api  # vérifier la taille de l'image

Étape 4 — docker-compose.prod.yml

Créer docker-compose.prod.yml :

services:
  api:
    build:
      context: .
      dockerfile: Dockerfile
      target: runner
    container_name: taskflow_api
    env_file: .env.production
    ports:
      - "3000:3000"
    depends_on:
      postgres:
        condition: service_healthy
    restart: unless-stopped
    networks:
      - taskflow_net

  postgres:
    image: postgres:16-alpine
    container_name: taskflow_postgres_prod
    environment:
      POSTGRES_DB: ${DB_NAME}
      POSTGRES_USER: ${DB_USER}
      POSTGRES_PASSWORD: ${DB_PASSWORD}
    volumes:
      - pgdata_prod:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${DB_USER}"]
      interval: 10s
      timeout: 5s
      retries: 5
    restart: unless-stopped
    networks:
      - taskflow_net

volumes:
  pgdata_prod:

networks:
  taskflow_net:
    driver: bridge

Créer .env.production (ne pas committer — l'ajouter au .gitignore) :

NODE_ENV=production
DB_HOST=postgres
DB_PORT=5432
DB_NAME=taskflow
DB_USER=taskflow_prod
DB_PASSWORD=CHANGER_MOT_DE_PASSE_FORT
JWT_SECRET=CHANGER_SECRET_LONG_ET_ALEATOIRE_MIN_64_CHARS
JWT_EXPIRES_IN=8h

Tester :

docker compose -f docker-compose.prod.yml up --build
curl http://localhost:3000/api/health

Étape 5 — Pipeline GitHub Actions

Créer .github/workflows/ci.yml :

name: CI TaskFlow API

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

env:
  NODE_VERSION: '20'

jobs:
  lint-and-test:
    name: Lint & Tests
    runs-on: ubuntu-latest

    services:
      postgres:
        image: postgres:16-alpine
        env:
          POSTGRES_DB: taskflow_test
          POSTGRES_USER: taskflow
          POSTGRES_PASSWORD: taskflow
        ports:
          - 5432:5432
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Node.js ${{ env.NODE_VERSION }}
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'npm'

      - name: Installer les dépendances
        run: npm ci

      - name: Linter ESLint
        run: npm run lint

      - name: Tests unitaires
        run: npm run test -- --coverage

      - name: Tests e2e
        run: npm run test:e2e
        env:
          NODE_ENV: test
          DB_HOST: localhost
          DB_PORT: 5432
          DB_NAME: taskflow_test
          DB_USER: taskflow
          DB_PASSWORD: taskflow
          JWT_SECRET: test-secret-key-for-ci
          JWT_EXPIRES_IN: 1h

  build-docker:
    name: Build Docker Image
    needs: lint-and-test
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main' && github.event_name == 'push'

    steps:
      - uses: actions/checkout@v4

      - name: Build image
        run: docker build -t taskflow-api:${{ github.sha }} .

      - name: Vérifier la taille de l'image
        run: docker images taskflow-api
